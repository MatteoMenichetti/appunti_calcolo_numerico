\makeglossaries

\newglossaryentry{equazioni lineari}{
	name = equazioni lineari,
	description = {Un'equazione lineare è un'equazione di primo grado, ovvero un'equazione un cui il grado massimo delle incognite è uguale ad uno ed è esprimibile come combinazione lineare delle incognite ed una costante. Le equazioni lineari sono del tipo $ax+b=0$, oppure del tipo $a_1x_1+\cdots+a_nx_n$ se in $n$ incognite}
}

\newglossaryentry{funzione lineare}{
	name = funzione lineare,
	description = {Nel calcolo infinitesimale, una funzione $f(x)$ è lineare se è una funzione polinomiale di grado zero o uno, del tipo $f(x)=mx+c$ con $m,c\in\mathbb R$ costanti reali. Se il coefficiente angolare (o gradiente) $m>0$ allora la funzione è strettamente crescente, se $m<0$ è strettamente decrescente. Nel piano cartesiano vengono visualizzate come equazioni del tipo $y=mx+c$. Il concetto può estendersi a funzioni di più variabili reali, ovvero: $f(x,y)=mx+ny+c$. Esempi: $f(x)=2x+1,\, f(x)=9$}
}

\newglossaryentry{maggiora uniformemente}{
	name = maggiora uniformemente,
	description={Una funzione $f\colon D\subseteq\mathbb R\rightarrow\mathbb{R}$ è maggiorata unifomemente se comunque fissato un valore $\varepsilon>0$ è possibile determinare $\delta=\delta(\varepsilon) > 0$ tale che $\forall x\in D$ che soddisfano $|x| < \delta$ allora $f(x) < \varepsilon$}
}

\newglossaryentry{range(A)}{
	name = $\range(A)$,
	description = {$\equiv\ran(A)=\{y\in\mathbb R^m:\exists x\in\mathbb R^n, y=Ax\}\, (\dim(\ran(A)=n))$}
}

\newglossaryentry{rank(A)}{
	name=$\rank(A)$,
	description={Massimo numero di righe (colonne) linearmente indipendenti}
}

\newglossaryentry{nonsingolare}{
	name=nonsingolare,
	description={$A\in\mathbb R^{n\times n}$ è una matrice nonsingolare se \gls{rank(A)}$=n$ ($\boldsymbol{\det(A)\neq 0}$) ed il sistema $A\uline x = b$ ha soluzione unica del tipo $\uline x = A^{-1} \uline b$. Vedere Definizione \ref{def:matrice_nonsingolare} di matrice nonsingolare}
}

\newglossaryentry{singolare}{
	name = singolare,
	description={$A\in\mathbb R^{n\times n}$ è una matrice \gls{singolare} se $\rank(A)\neq n$, quindi $\det(A) = 0$. Vedere Definizione \ref{def:matrice_singolare} di matrice singolare}
}

\newglossaryentry{metodo lineare}{
	name = metodo lineare,
	description={Un metodo iterativo è lineare se il suo ordine di convergenza è $p=1$}
}

\newglossaryentry{matrice identita}{
	name = matrice identità,
	description={La matrice identità è una matrice quadrata in cui tutti gli elementi della diagonale sono 1 mentre i restanti 0. È indicata con $I$ oppure $I_n$ ed è rappresentabule come
	\begin{equation*}
		 I_{1} = 
		 \begin{bmatrix}
		 	1
		 \end{bmatrix}\quad
		 I_2 = 
		 \begin{bmatrix}
		 	1&0\\
		 	0&1
	 	\end{bmatrix}\quad 
	 	I_n =
	 	\begin{bmatrix}
	 		1&0&\cdots &0\\
	 		0&1&\cdots &0\\
	 		\vdots &\vdots &\ddots &\vdots \\
	 		0&0&\cdots &1
 		\end{bmatrix}
	\end{equation*}}
}

\newglossaryentry{metodo numerico}{
	name = metodo numerico,
	description = {Un metodo numerico è un metodo di calcolo di problema matematico del tipo $y=f(x)$, dove il risultato ottenuto dal metodo è un approssimazione del problema. Un metodo numerico è caratterizzato da un algoritmo e di conseguenza dalla sua stabilità, convergenza e computabilità}
}

\newglossaryentry{sistemi lineari}{
	name = sistemi lineari,
	description={Sistemi di \gls{equazioni lineari} della forma $A\uline x = \uline b$, dove $A=(a_{ij})\in{\mathbb R^{m\times n}}$ è la matrice dei coefficienti, $\uline x=(x_i)\in\mathbb R^n$ il vettore delle incognite (da determinare) e $\uline b\in\mathbb R^m$ il vettore dei termini noti}
}

\newglossaryentry{equazioni nonlineari}{
	name = equazioni nonlineari,
	description={Un'equazione nonlineare è un'equazione di grado maggiore del primo, ovvero un'equazione un cui il grado massimo delle incognite è maggiore di uno e non è esprimibile come combinazione lineare delle incognite ed una costante}
}

\newglossaryentry{forma component-wise}{
	name = forma component-wise,
	description={Con \textit{forma component-wise} si intende una forma estesa, componente per componente di un oggetto matematico}
}

\newglossaryentry{ben posto}{
	name = ben posto,
	description={Riferito ad un qualsiasi problema matematico significa che la soluzione esiste ed è unica}
}

\newglossaryentry{ben definito}{
	name=ben definito,
	description={Sinonimo di \gls{ben posto}, relativo ad espressioni. Ovvero: ad una espressione, come una funzione,assegnata un'unica interpretazione o valore senza ambiguità}
}

\newglossaryentry{polinomio monico}{
	name=polinomio monico,
	description={Un polinomio $p(x)\in\Pi_n$ del tipo
		\begin{equation*}
			p(x) = a_n x^n + \hdots + a_0 x^0,
		\end{equation*}
		si dice monico se $a_n=1$}
}
